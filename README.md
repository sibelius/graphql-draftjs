# GraphQL Draftjs

GraphQL types for DraftJS

## Usage
```jsx
import { 
  DraftContentStateType,
  DraftContentStateInput,
  DraftBlock,
  DraftBlockInput,
  DraftEntity,
  DraftEntityInput,
  DraftData,
  DraftDataInput,
  DraftEntityRange,
  DraftEntityRangeInput,
  DraftInlineStyleRange,
  DraftInlineStyleRangeInput,
} from 'draftjs-graphql'
```
